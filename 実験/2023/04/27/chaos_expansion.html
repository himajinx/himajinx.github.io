変分法からカオス展開へ

Apr 27, 2023

後日追記します
最近、カオス展開がとても気になっている。
微分方程式のシュミレーションに使うらしいが、日常でも使えないものだろうか？

実験してみた。

var trueMu = 3.5
var trueSigma = 0.8

var h1=(function(x){1})
var h2=(function(x){x})
var h3=(function(x){x*x-1})
var h4=(function(x){x*x*x-3*x})

var data = repeat(1000, function() { return (function(v){v*v+1})(gaussian(trueMu, trueSigma))})

var model=function(){
  var p1=param("p1")
  var p2=param("p2")
  var p3=param("p3")
  var p4=param("p4")
  
  print([p1,p2,p3,p4])


  var tgt=function(p){return p1*h1(p)+p2*h2(p)+p3*h3(p)+p4*h4(p)}
  var smpl=repeat(1000,function(){tgt(Gaussian({mu:0,sigma:1}).sample())})
  //print(smpl)
  factor(Gaussian({mu:listMean(smpl),sigma:3}).score(listMean(data)))
  factor(Gaussian({mu:Math.sqrt(listVar(smpl)),sigma:3}).score(listMean(data)))
}
//var vv=Optimize({method: 'MCMC', samples: 10000}, model);


Optimize({
  model:  model,
  optMethod: {adam: {stepSize: .25}},
  steps: 250,
  verbose:true}
) <br>

結論:

  • 定義域が上手く再現出来ていない
  • 目的関数をどうするか悩む(パラメータなら出力分布があるが、今回は？？？）

追記

  1. サンプルの正規化
  2. モーメントの個数を調整

この2つを加えたら上手く行った

さらに追記

-　ベイズ更新でカオス展開を活用する場合、もう少し勉強する必要がある。
特に、展開するランダム変数が3つ以上の場合、期待値の計算がかなり怪しい。
